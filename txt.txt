void main() async {
  WidgetsFlutterBinding.ensureInitialized();

  // Configure window for desktop platforms (macOS, Windows, Linux)
  if (Platform.isMacOS || Platform.isWindows || Platform.isLinux) {
    await windowManager.ensureInitialized();

    WindowOptions windowOptions = const WindowOptions(
      size: Size(1920, 1080), // Set a large default size
      center: true,
      backgroundColor: Colors.transparent,
      skipTaskbar: false,
      titleBarStyle: TitleBarStyle.normal,
      fullScreen: false, // Set to true if you want truly full screen
    );

    await windowManager.waitUntilReadyToShow(windowOptions, () async {
      await windowManager.show();
      await windowManager.focus();
      await windowManager.maximize(); // Maximize the window
      await windowManager.setResizable(false); // Prevent resizing
      await windowManager.setMaximizable(false); // Disable maximize button
      await windowManager.setMinimizable(true); // Allow minimize
    });
  }

  runApp(const MaderaKitchenApp());
}

      print('⚠️ Error during cleanup: $e');

enum AppMode { notConfigured, master, slave }

class AppModeProvider extends ChangeNotifier {
  AppMode _mode = AppMode.notConfigured;
  AppMode get mode => _mode;

  bool _isInitialized = false;
  bool get isInitialized => _isInitialized;

  bool _isStartingBackend = false;
  bool get isStartingBackend => _isStartingBackend;

  bool _waitingForSlaves = false;
  bool get waitingForSlaves => _waitingForSlaves;

  bool _waitingForMasterReady = false;
  bool get waitingForMasterReady => _waitingForMasterReady;

  String? _masterIp;
  String? get masterIp => _masterIp;

  String? _errorMessage;
  String? get errorMessage => _errorMessage;

  String _statusMessage = '';
  String get statusMessage => _statusMessage;

  final Set<Map<String, String>> _connectedSlaves = {};
  Set<Map<String, String>> get connectedSlaves => _connectedSlaves;
  int get connectedSlaveCount => _connectedSlaves.length;

  final NetworkDiscoveryService _networkService = NetworkDiscoveryService();
  BackendManagerService? _backendManager;

  static const String _modeKey = 'app_mode';
  static const String _backendPathKey = 'backend_path';

  Future<void> initialize() async {
    if (_isInitialized) return;

    try {
      print('🔧 Initializing AppModeProvider...');
      _statusMessage = 'Initializing application...';
      notifyListeners();

      final prefs = await SharedPreferences.getInstance();
      final savedMode = prefs.getString(_modeKey);

      if (savedMode != null) {
        _mode = AppMode.values.firstWhere(
          (e) => e.toString() == savedMode,
          orElse: () => AppMode.notConfigured,
        );
        print('📱 Loaded saved mode: $_mode');

        if (_mode == AppMode.master) {
          print('🚀 AUTO-STARTING MASTER MODE');
          _statusMessage = 'Starting master mode...';
          notifyListeners();
          await _startMasterMode();
        } else if (_mode == AppMode.slave) {
          print('🚀 AUTO-STARTING SLAVE MODE');
          _statusMessage = 'Starting slave mode...';
          notifyListeners();
          await _startSlaveMode();
        }
      }

      _isInitialized = true;
      _statusMessage = 'Ready';
      notifyListeners();
    } catch (e) {
      print('❌ Initialization error: $e');
      _errorMessage = 'Failed to initialize: $e';
      _statusMessage = 'Initialization failed';
      notifyListeners();
    }
  }

  Future<void> setMasterMode(String backendPath) async {
    try {
      print('🎯 Setting up MASTER mode...');
      _isStartingBackend = true;
      _errorMessage = null;
      _statusMessage = 'Starting Django backend server...';
      notifyListeners();

      _backendManager = BackendManagerService(backendPath: backendPath);

      // Start backend and wait for it to be confirmed running
      final serverStarted = await _backendManager!.startBackend(host: '0.0.0.0', port: 8000);
     
      if (!serverStarted) {
        throw Exception('Django backend failed to start or is not responding');
      }
     
      print('✅ DJANGO BACKEND LAUNCHED AND CONFIRMED RUNNING');

      // Start network broadcast (server runs regardless of slaves)
      _statusMessage = 'Starting network broadcast...';
      notifyListeners();
      await _networkService.startMasterBroadcast();
      _masterIp = _networkService.currentMasterIp;

      // Listen for slave acceptances (optional - server runs even with 0 slaves)
      _networkService.slaveAcceptStream.listen((data) {
        _handleSlaveAcceptance(data);
      });

      // Update backend URL ONLY after server is confirmed running
      _updateBackendUrl();
      print('✅ Backend URL updated to: ${_networkService.getBackendUrl()}');

      _mode = AppMode.master;
      final prefs = await SharedPreferences.getInstance();
      await prefs.setString(_modeKey, _mode.toString());
      await prefs.setString(_backendPathKey, backendPath);

      _isStartingBackend = false;
      _waitingForSlaves = true;
      _statusMessage = 'Server running - Waiting for slaves (optional)...';
      print('✅ Master mode active - Server is running (0 slaves required)');
      notifyListeners();
    } catch (e) {
      print('❌ Failed to set master mode: $e');
      _errorMessage = 'Failed to start master mode: $e';
      _statusMessage = 'Master mode failed';
      _mode = AppMode.notConfigured;
      _isStartingBackend = false;
      notifyListeners();
      rethrow;
    }
  }

  void _handleSlaveAcceptance(Map<String, dynamic> data) {
    final slaveId = data['slave_id'] as String;
    final slaveIp = data['slave_ip'] as String;

    _connectedSlaves.add({
      'id': slaveId,
      'ip': slaveIp,
    });

    _statusMessage = 'Slaves connected: ${_connectedSlaves.length}';
    print('✅ Slave accepted: $slaveId at $slaveIp');
    notifyListeners();
  }

  void confirmSlaveConfiguration() {
    print('✅ Master confirming slave configuration...');
    _networkService.sendMasterReady();
    _waitingForSlaves = false;
    _statusMessage = 'Configuration confirmed';
    notifyListeners();
  }

  Future<void> setSlaveMode() async {
    try {
      print('🎯 Setting up SLAVE mode...');
      _errorMessage = null;
      _statusMessage = 'Discovering master server...';
      notifyListeners();

      await _networkService.startSlaveDiscovery();

      _networkService.masterIpStream.listen((ip) {
        _masterIp = ip;
        _statusMessage = 'Master found at $ip';
        notifyListeners();
      });

      _networkService.masterReadyStream.listen((ready) {
        if (ready) {
          _waitingForMasterReady = false;
          _statusMessage = 'Connected to master';
          _updateBackendUrl();
          notifyListeners();
        }
      });

      _mode = AppMode.slave;
      final prefs = await SharedPreferences.getInstance();
      await prefs.setString(_modeKey, _mode.toString());

      _statusMessage = 'Listening for master...';
      print('✅ Slave mode configured');
      notifyListeners();
    } catch (e) {
      print('❌ Failed to set slave mode: $e');
      _errorMessage = 'Failed to start slave mode: $e';
      _statusMessage = 'Slave mode failed';
      _mode = AppMode.notConfigured;
      notifyListeners();
      rethrow;
    }
  }

  Future<void> acceptMasterIp() async {
    try {
      print('✅ Accepting master IP: $_masterIp');

      _waitingForMasterReady = true;
      _statusMessage = 'Waiting for master confirmation...';
      notifyListeners();

      // Get local IP for slave identification
      final interfaces = await NetworkInterface.list();
      String? slaveIp;

      for (var interface in interfaces) {
        for (var addr in interface.addresses) {
          if (addr.type == InternetAddressType.IPv4 && !addr.isLoopback) {
            slaveIp = addr.address;
            break;
          }
        }
      }

      final slaveId = Platform.localHostname;
      await _networkService.sendSlaveAcceptance(slaveId, slaveIp ?? 'unknown');
     
      print('✅ Slave acceptance sent');
    } catch (e) {
      print('❌ Failed to accept master IP: $e');
      _errorMessage = 'Failed to accept configuration: $e';
      notifyListeners();
      rethrow;
    }
  }

  Future<void> _startMasterMode() async {
    final prefs = await SharedPreferences.getInstance();
    final backendPath = prefs.getString(_backendPathKey);

    if (backendPath != null) {
      print('🔄 Found saved backend path: $backendPath');
      await setMasterMode(backendPath);
    } else {
      print('⚠️ Backend path not found');
      _mode = AppMode.notConfigured;
      notifyListeners();
    }
  }

  Future<void> _startSlaveMode() async {
    await setSlaveMode();
  }

  void _updateBackendUrl() {
    final backendUrl = _networkService.getBackendUrl();
    print('🔄 Updating backend URL to: $backendUrl');
    DioClient().updateBaseUrl(backendUrl);
  }

  String getBackendUrl() {
    return _networkService.getBackendUrl();
  }

  Future<void> resetMode() async {
    print('🔄 Resetting app mode...');

    _networkService.stop();
    await _backendManager?.stopBackend();

    final prefs = await SharedPreferences.getInstance();
    await prefs.remove(_modeKey);
    await prefs.remove(_backendPathKey);

    _mode = AppMode.notConfigured;
    _masterIp = null;
    _errorMessage = null;
    _backendManager = null;
    _connectedSlaves.clear();
    _waitingForSlaves = false;
    _waitingForMasterReady = false;

    notifyListeners();
  }

  @override
  void dispose() {
    print('🧹 Cleaning up AppModeProvider...');
    _networkService.dispose();
    _backendManager?.dispose();
    super.dispose();
  }
}

/// Service to manage Django backend server lifecycle
/// Master starts/stops the backend automatically
class BackendManagerService {
  Process? _djangoProcess;
  bool _isRunning = false;
  bool get isRunning => _isRunning;

  final String _backendPath;
  final String _pythonExecutable;

  BackendManagerService({
    required String backendPath,
    String pythonExecutable = 'python3',
  }) : _backendPath = backendPath,
       _pythonExecutable = pythonExecutable;

  // ═══════════════════════════════════════════════════════════════
  // START DJANGO BACKEND SERVER
  // ═══════════════════════════════════════════════════════════════

  Future<bool> startBackend({String host = '0.0.0.0', int port = 8000}) async {
    if (_isRunning) {
      print('⚠️ Backend is already running');
      return true;
    }

    try {
      print('🚀 Starting Django backend server...');
      print('📂 Backend path: $_backendPath');
      print('🐍 Python executable: $_pythonExecutable');

      // Verify backend path exists
      final backendDir = Directory(_backendPath);
      if (!await backendDir.exists()) {
        throw Exception('Backend directory not found: $_backendPath');
      }

      // Use the madira directory as working directory (where the command should run)
      final workingDir = _backendPath;
     
      print('🎯 Working directory: $workingDir');
      print('🎯 Starting server at: $host:$port');
      print('📁 Changing to directory: $workingDir');

      // Try to find uvicorn path first, then use it directly (as it works in terminal)
      // This avoids macOS security restrictions
      String? uvicornPath;
      if (Platform.isMacOS) {
        try {
          print('🔍 Looking for uvicorn in PATH...');
          final whichResult = await Process.run(
            'zsh',
            ['-l', '-c', 'export PATH="/opt/homebrew/bin:/usr/local/bin:\$PATH" && which uvicorn'],
            runInShell: false,
          );
          print('🔍 which uvicorn exit code: ${whichResult.exitCode}');
          print('🔍 which uvicorn stdout: ${whichResult.stdout}');
          print('🔍 which uvicorn stderr: ${whichResult.stderr}');
          if (whichResult.exitCode == 0 && whichResult.stdout.toString().trim().isNotEmpty) {
            uvicornPath = whichResult.stdout.toString().trim();
            print('✅ Found uvicorn at: $uvicornPath');
          } else {
            print('⚠️ uvicorn not found in PATH, will use python3 -m uvicorn');
          }
        } catch (e) {
          print('⚠️ Could not find uvicorn path: $e');
          print('⚠️ Will use python3 -m uvicorn instead');
        }
      }

      // Build command - use uvicorn directly if found, otherwise use python3 -m uvicorn
      final command = uvicornPath != null
          ? '$uvicornPath madira.asgi:application --host $host --port $port --workers 4'
          : '$_pythonExecutable -m uvicorn madira.asgi:application --host $host --port $port --workers 4';
     
      print('🎯 Full command: $command');
      print('🎯 Python executable: $_pythonExecutable');
      print('🎯 Running from directory: $workingDir');
     
      // On macOS, use zsh and source profile to get proper PATH (especially for homebrew)
      final shell = Platform.isWindows
          ? 'cmd'
          : Platform.isMacOS
              ? 'zsh'
              : 'bash';
     
      // Build shell command - use login shell on macOS to get proper PATH
      // Add explicit PATH setup for macOS to ensure python3/uvicorn is found
      // Escape $PATH with \$ so it's treated as shell variable, not Dart variable
      final shellCommand = Platform.isWindows
          ? 'cd /d "$workingDir" && $command'
          : Platform.isMacOS
              ? 'export PATH="/opt/homebrew/bin:/usr/local/bin:\$PATH" && cd "$workingDir" && pwd && $command'
              : 'cd "$workingDir" && $command';
     
      // Use login shell on macOS to automatically source profile and get PATH
      final shellArgs = Platform.isWindows
          ? ['/c', shellCommand]
          : Platform.isMacOS
              ? ['-l', '-c', shellCommand]  // -l makes it a login shell
              : ['-c', shellCommand];
     
      print('🎯 Shell: $shell');
      print('🎯 Shell command: $shellCommand');
      print('🎯 Shell args: $shellArgs');
     
      _djangoProcess = await Process.start(
        shell,
        shellArgs,
        runInShell: true,
      );

      _isRunning = true;
      final processPid = _djangoProcess!.pid;
      print('🔄 Django process started with PID: $processPid');

      // Buffer to collect output for debugging
      final outputBuffer = StringBuffer();
      final errorBuffer = StringBuffer();
      bool processExited = false;
      int? exitCode;

      // Listen to output
      _djangoProcess!.stdout
          .transform(SystemEncoding().decoder)
          .transform(const LineSplitter())
          .listen((line) {
        if (line.trim().isNotEmpty) {
          print('🔹 Django: $line');
          outputBuffer.writeln(line);
        }
      });

      _djangoProcess!.stderr
          .transform(SystemEncoding().decoder)
          .transform(const LineSplitter())
          .listen((line) {
        if (line.trim().isNotEmpty) {
          print('🔸 Django Error: $line');
          errorBuffer.writeln(line);
        }
      });

      // Monitor process exit
      _djangoProcess!.exitCode.then((code) {
        exitCode = code;
        processExited = true;
        if (code != 0) {
          print('⚠️ Django server exited with code: $code');
          if (errorBuffer.isNotEmpty) {
            print('🔸 Full error output:\n${errorBuffer.toString()}');
          }
          if (outputBuffer.isNotEmpty) {
            print('🔹 Full output:\n${outputBuffer.toString()}');
          }
        }
        _isRunning = false;
        _djangoProcess = null;
      });

      // Give Django a moment to start and show initial output
      print('⏳ Waiting 3 seconds for Django to initialize...');
      await Future.delayed(Duration(seconds: 3));

      // Check if process already exited
      if (processExited) {
        print('❌ Django process exited immediately with code: $exitCode');
        if (errorBuffer.isNotEmpty) {
          print('🔸 Error output:\n${errorBuffer.toString()}');
        }
        if (outputBuffer.isEmpty) {
          print('⚠️ No output from Django process - command may have failed silently');
        }
        _isRunning = false;
        _djangoProcess = null;
        return false;
      }

      // Wait and verify server is responding with retries
      print('⏳ Waiting for Django to start and verify it\'s responding...');
      bool serverReady = false;
      for (int i = 0; i < 15; i++) {
        // Check if process exited during wait
        if (processExited) {
          print('❌ Django process exited during startup (code: $exitCode)');
          if (errorBuffer.isNotEmpty) {
            print('🔸 Error output:\n${errorBuffer.toString()}');
          }
          _isRunning = false;
          _djangoProcess = null;
          return false;
        }

        await Future.delayed(Duration(seconds: 2));
        if (await _isServerResponding(host, port)) {
          serverReady = true;
          print('✅ Django backend started successfully and responding at http://$host:$port');
          break;
        }
        print('⏳ Waiting for server to respond... (attempt ${i + 1}/15)');
      }

      if (!serverReady) {
        print('❌ Django backend process started but not responding on port $port');
        if (processExited) {
          print('❌ Process exited with code: $exitCode');
        }
        if (errorBuffer.isNotEmpty) {
          print('🔸 Error output:\n${errorBuffer.toString()}');
        }
        if (outputBuffer.isNotEmpty) {
          print('🔹 Output so far:\n${outputBuffer.toString()}');
        } else {
          print('⚠️ No output from Django - process may not have started correctly');
        }
        _isRunning = false;
        _djangoProcess?.kill();
        _djangoProcess = null;
        return false;
      }

      return true; // Server is running and responding
    } catch (e) {
      _isRunning = false;
      _djangoProcess?.kill();
      _djangoProcess = null;
      print('❌ Failed to start backend: $e');
      return false;
    }
  }

  // ═══════════════════════════════════════════════════════════════
  // STOP DJANGO BACKEND SERVER
  // ═══════════════════════════════════════════════════════════════

  Future<void> stopBackend() async {
    if (!_isRunning || _djangoProcess == null) {
      print('⚠️ Backend is not running');
      return;
    }

    try {
      print('🛑 Stopping Django backend server...');

      // Try graceful shutdown first
      _djangoProcess!.kill(ProcessSignal.sigterm);

      // Wait up to 5 seconds for graceful shutdown
      final timeout = Future.delayed(Duration(seconds: 5));
      final exitCode = await Future.any([
        _djangoProcess!.exitCode,
        timeout.then((_) => -1),
      ]);

      if (exitCode == -1) {
        // Force kill if graceful shutdown failed
        print('⚠️ Graceful shutdown timeout, forcing kill...');
        _djangoProcess!.kill(ProcessSignal.sigkill);
      }

      _isRunning = false;
      _djangoProcess = null;

      print('✅ Django backend stopped');
    } catch (e) {
      print('❌ Error stopping backend: $e');
      _isRunning = false;
      _djangoProcess = null;
    }
  }

  // ═══════════════════════════════════════════════════════════════
  // UTILITY METHODS
  // ═══════════════════════════════════════════════════════════════

  Future<bool> _isServerResponding(String host, int port) async {
    try {
      // First check if port is open
      final socket = await Socket.connect(
        host == '0.0.0.0' ? '127.0.0.1' : host,
        port,
        timeout: Duration(seconds: 2),
      );
      socket.destroy();
     
      // Then verify HTTP endpoint is actually responding
      final httpClient = HttpClient();
      final uri = Uri.parse('http://${host == '0.0.0.0' ? '127.0.0.1' : host}:$port/api/dashboard/');
     
      try {
        final request = await httpClient.getUrl(uri).timeout(Duration(seconds: 3));
        final response = await request.close().timeout(Duration(seconds: 3));
        httpClient.close();
       
        // Server is responding if we get any HTTP response (even 401/403 is OK - means server is up)
        final isResponding = response.statusCode >= 200 && response.statusCode < 500;
        if (isResponding) {
          print('✅ Server HTTP endpoint responding (status: ${response.statusCode})');
        }
        return isResponding;
      } catch (e) {
        httpClient.close();
        return false;
      }
    } catch (e) {
      return false;
    }
  }

  Future<bool> restartBackend({
    String host = '0.0.0.0',
    int port = 8000,
  }) async {
    print('🔄 Restarting Django backend...');
    await stopBackend();
    await Future.delayed(Duration(seconds: 2));
    return await startBackend(host: host, port: port);
  }

  // ═══════════════════════════════════════════════════════════════
  // CLEANUP
  // ═══════════════════════════════════════════════════════════════

  Future<void> dispose() async {
    await stopBackend();
  }
}

class NetworkDiscoveryService {
  static const int BROADCAST_PORT = 8888;
  static const int BACKEND_PORT = 8000;
  static const String DISCOVERY_MESSAGE = 'MADIRA_MASTER_SERVER';
  static const String SLAVE_ACCEPT_MESSAGE = 'MADIRA_SLAVE_ACCEPT';
  static const String MASTER_READY_MESSAGE = 'MADIRA_MASTER_READY';

  RawDatagramSocket? _broadcastSocket;
  RawDatagramSocket? _listenSocket;
  Timer? _broadcastTimer;

  final _masterIpController = StreamController<String>.broadcast();
  Stream<String> get masterIpStream => _masterIpController.stream;

  final _slaveAcceptController = StreamController<Map<String, dynamic>>.broadcast();
  Stream<Map<String, dynamic>> get slaveAcceptStream => _slaveAcceptController.stream;

  final _masterReadyController = StreamController<bool>.broadcast();
  Stream<bool> get masterReadyStream => _masterReadyController.stream;

  String? _currentMasterIp;
  String? get currentMasterIp => _currentMasterIp;

  // Master broadcasts IP
  Future<void> startMasterBroadcast() async {
    try {
      print('📡 Starting MASTER broadcast mode...');

      final localIp = await _getLocalIpAddress();
      if (localIp == null) {
        throw Exception('Could not determine local IP address');
      }

      print('📡 Master IP: $localIp');
      _currentMasterIp = localIp;

      _broadcastSocket = await RawDatagramSocket.bind(
        InternetAddress.anyIPv4,
        0,
      );
      _broadcastSocket!.broadcastEnabled = true;

      // Listen for slave acceptance messages
      _listenSocket = await RawDatagramSocket.bind(
        InternetAddress.anyIPv4,
        BROADCAST_PORT + 1,
      );

      _listenSocket!.listen((event) {
        if (event == RawSocketEvent.read) {
          final datagram = _listenSocket!.receive();
          if (datagram != null) {
            _handleSlaveAcceptMessage(datagram);
          }
        }
      });

      // Broadcast every 3 seconds
      _broadcastTimer = Timer.periodic(Duration(seconds: 3), (timer) {
        _sendBroadcast(localIp);
      });

      _sendBroadcast(localIp);

      print('✅ Master broadcast started on port $BROADCAST_PORT');
    } catch (e) {
      print('❌ Failed to start master broadcast: $e');
      rethrow;
    }
  }

  void _sendBroadcast(String ipAddress) {
    try {
      final message = jsonEncode({
        'type': DISCOVERY_MESSAGE,
        'ip': ipAddress,
        'port': BACKEND_PORT,
        'timestamp': DateTime.now().millisecondsSinceEpoch,
      });

      final data = utf8.encode(message);
      _broadcastSocket?.send(
        data,
        InternetAddress('255.255.255.255'),
        BROADCAST_PORT,
      );

      print('📤 Broadcast sent: $ipAddress');
    } catch (e) {
      print('❌ Broadcast send error: $e');
    }
  }

  void _handleSlaveAcceptMessage(Datagram datagram) {
    try {
      final message = utf8.decode(datagram.data);
      final data = jsonDecode(message);

      if (data['type'] == SLAVE_ACCEPT_MESSAGE) {
        print('✅ Slave acceptance received: ${data['slave_id']}');
        _slaveAcceptController.add(data);
      }
    } catch (e) {
      print('❌ Error processing slave acceptance: $e');
    }
  }

  // Slave listens for master broadcasts
  Future<void> startSlaveDiscovery() async {
    try {
      print('👂 Starting SLAVE discovery mode...');

      _listenSocket = await RawDatagramSocket.bind(
        InternetAddress.anyIPv4,
        BROADCAST_PORT,
      );

      _listenSocket!.listen((event) {
        if (event == RawSocketEvent.read) {
          final datagram = _listenSocket!.receive();
          if (datagram != null) {
            _handleBroadcastMessage(datagram);
          }
        }
      });

      print('✅ Slave discovery started, listening on port $BROADCAST_PORT');
    } catch (e) {
      print('❌ Failed to start slave discovery: $e');
      rethrow;
    }
  }

  void _handleBroadcastMessage(Datagram datagram) {
    try {
      final message = utf8.decode(datagram.data);
      final data = jsonDecode(message);

      if (data['type'] == DISCOVERY_MESSAGE) {
        final masterIp = data['ip'] as String;
        final masterPort = data['port'] as int;

        if (_currentMasterIp != masterIp) {
          _currentMasterIp = masterIp;
          _masterIpController.add(masterIp);
          print('🎯 Master discovered at: $masterIp:$masterPort');
        }
      } else if (data['type'] == MASTER_READY_MESSAGE) {
        print('✅ Master confirmed configuration');
        _masterReadyController.add(true);
      }
    } catch (e) {
      print('❌ Error processing broadcast message: $e');
    }
  }

  // Slave sends acceptance to master
  Future<void> sendSlaveAcceptance(String slaveId, String slaveIp) async {
    try {
      if (_currentMasterIp == null) {
        throw Exception('Master IP not known');
      }

      final socket = await RawDatagramSocket.bind(
        InternetAddress.anyIPv4,
        0,
      );

      final message = jsonEncode({
        'type': SLAVE_ACCEPT_MESSAGE,
        'slave_id': slaveId,
        'slave_ip': slaveIp,
        'timestamp': DateTime.now().millisecondsSinceEpoch,
      });

      final data = utf8.encode(message);
      socket.send(
        data,
        InternetAddress(_currentMasterIp!),
        BROADCAST_PORT + 1,
      );

      print('📤 Slave acceptance sent to master: $slaveId');
      socket.close();
    } catch (e) {
      print('❌ Failed to send slave acceptance: $e');
      rethrow;
    }
  }

  // Master sends ready confirmation
  void sendMasterReady() {
    try {
      final message = jsonEncode({
        'type': MASTER_READY_MESSAGE,
        'timestamp': DateTime.now().millisecondsSinceEpoch,
      });

      final data = utf8.encode(message);
      _broadcastSocket?.send(
        data,
        InternetAddress('255.255.255.255'),
        BROADCAST_PORT,
      );

      print('📤 Master ready broadcast sent');
    } catch (e) {
      print('❌ Master ready send error: $e');
    }
  }

  Future<String?> _getLocalIpAddress() async {
    try {
      final info = NetworkInfo();

      String? ip = await info.getWifiIP();
      if (ip != null && ip.isNotEmpty && ip != '127.0.0.1') {
        return ip;
      }

      for (var interface in await NetworkInterface.list()) {
        for (var addr in interface.addresses) {
          if (addr.type == InternetAddressType.IPv4 &&
              !addr.isLoopback &&
              !addr.address.startsWith('169.254')) {
            return addr.address;
          }
        }
      }

      return null;
    } catch (e) {
      print('❌ Error getting local IP: $e');
      return null;
    }
  }

  String getBackendUrl() {
    if (_currentMasterIp != null) {
      return 'http://$_currentMasterIp:$BACKEND_PORT';
    }
    return 'http://127.0.0.1:$BACKEND_PORT';
  }

  void stop() {
    print('🛑 Stopping network discovery...');
    _broadcastTimer?.cancel();
    _broadcastSocket?.close();
    _listenSocket?.close();
    _currentMasterIp = null;
  }

  void dispose() {
    stop();
    _masterIpController.close();
    _slaveAcceptController.close();
    _masterReadyController.close();
  }
}

                  'Enter the path to your Django backend directory\n(the folder containing manage.py)',
                ),
                const SizedBox(height: 16),
                TextField(
                  controller: controller,
                  decoration: InputDecoration(
                    labelText: 'Backend Path',
                    border: const OutlineInputBorder(),
                    suffixIcon: IconButton(
                      icon: const Icon(Icons.folder_open),
                      onPressed: () async {
                        final result =
                            await FilePicker.platform.getDirectoryPath();
                        if (result != null) {
                          controller.text = result;
                        }
                      },
                    ),
                  ),
                ),
              ],
            ),
            actions: [
              TextButton(
                onPressed: () => Navigator.pop(context),
                child: const Text('Cancel'),
              ),
              ElevatedButton(
                onPressed: () {
                  if (controller.text.isNotEmpty) {
                    Navigator.pop(context, controller.text);
                  }
                },
                child: const Text('Confirm'),
              ),
            ],
          ),
    );
  }
